CREATE DATABASE cc_model_db;

USE cc_model_db


CREATE TABLE user (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(20),
    password VARCHAR(20),
    create_date DATE,
    email VARCHAR(100)
);

CREATE TABLE class_group (
    id INT AUTO_INCREMENT PRIMARY KEY,
    group_name VARCHAR(20),
    userId INT,
    FOREIGN KEY (userID) REFERENCES user (id)
);

CREATE TABLE group_scenario (
    id INT AUTO_INCREMENT PRIMARY KEY,
    yearId INT, 
    FOREIGN KEY (yearId) REFERENCES year(id),
    decisionId INT, 
    FOREIGN KEY (decisionId) REFERENCES decision(id),
    total INT,
    income INT,
    cost INT,
    net INT
);

CREATE TABLE decision (
    id INT AUTO_INCREMENT PRIMARY KEY,
    decision INT
);

CREATE TABLE year (
    id INT AUTO_INCREMENT PRIMARY KEY,
    year INT
);

CREATE TABLE scenario (
    id INT AUTO_INCREMENT PRIMARY KEY,
    userId INT, 
    FOREIGN KEY (userId) REFERENCES user(id),
    yearId INT, 
    FOREIGN KEY (yearId) REFERENCES year(id),
    disaster_scenarioId INT,
    FOREIGN KEY (disaster_scenarioId) REFERENCES disaster_scenario(id),
    emissions_growth FLOAT,
    co2_ppm INT,
    co2_radiative_forcing FLOAT,
    temp_increase FLOAT,
    ocean_heat_storage FLOAT,    
    original_disaster_risk INT
);

CREATE TABLE disaster_scenario (
    id INT AUTO_INCREMENT PRIMARY KEY,
    probability INT,
    decision1 INT,
    decision2 INT,
    decision3 INT,
    decision4 INT,
    decision5 INT
);
